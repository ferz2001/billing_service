from datetime import datetime

from pydantic import BaseModel, RootModel


class MovieEvent(BaseModel):
    movie_timestamp: int
    movie_id: str
    user_id: str
    event_time: int = int(datetime.now().timestamp())


class MovieEventList(RootModel):
    root: list[MovieEvent]


data_models = {
    "events": MovieEventList,
}
