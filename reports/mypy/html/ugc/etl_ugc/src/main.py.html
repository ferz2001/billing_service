import logging
from logging import config as logging_config
from time import sleep

import orjson
import redis
from pydantic import RootModel
from client.clickhouse_client import ClickHouseColumnDBClient
from client.kafka_client import KafkaMessageBrokerConsumer
from core.config import (
    clickhouse_settings,
    etl_settings,
    kafka_settings,
    redis_settings,
)
from core.logger import LOGGING
from extractor import Extractor
from helpers.graceful_kill import GracefulKiller
from loader import Loader
from models.events import data_models
from state_redis import RedisStorage, State

logging_config.dictConfig(LOGGING)
logger = logging.getLogger(__name__)


class ETLProcess:
    def __init__(self):
        self.redis_client = self.create_redis_client()
        self.db_client = self.create_db_client()
        self.message_broker = self.create_message_broker()
        self.killer = GracefulKiller()

    def create_redis_client(self):
        return redis.Redis(
            host=redis_settings.host, port=redis_settings.port, decode_responses=True,
        )

    def create_db_client(self):
        clickhouse_dsn = {
            "host": clickhouse_settings.host,
            "port": clickhouse_settings.port,
            "user": clickhouse_settings.user,
            "password": clickhouse_settings.password,
            "database": clickhouse_settings.database,
        }
        db_client = ClickHouseColumnDBClient(**clickhouse_dsn)
        db_client.connect()
        return db_client

    def create_message_broker(self):
        kafka_host, kafka_port = kafka_settings.bootstrap_servers[0].split(":")
        logger.info(f'@@@@@@@@@@@@@   Kafka host: {kafka_host}, port: {kafka_port}   @@@@@@@@@@@@@@@@@@@@@')
        message_broker = KafkaMessageBrokerConsumer()
        message_broker.connect(
            kafka_host,
            kafka_port,
            etl_settings.broker_identifier,
            batch_size=etl_settings.batch_size,
            value_deserializer=orjson.loads,
        )
        message_broker.subscribe(kafka_settings.topics)
        return message_broker

    def run_etl(self):
        storage = RedisStorage(self.redis_client)
        state_instance = State(storage)
        extractor = Extractor(self.message_broker)
        loader = Loader(self.db_client)

        data_batch = state_instance.get_state("event_etl_data")

        if not data_batch:
            data_batch = extractor.extract()
            state_instance.set_state("event_etl_data", data_batch)
            logger.debug("Extracting data")
        else:
            logger.warning("Using cached data! %s", data_batch)

        if data_batch:
            for data_type in data_batch:
                data_model: RootModel = data_models[data_type]
                logger.debug(f'DataModel: {data_model}')
                topic_data = data_model.model_validate(data_batch[data_type])
                logger.info(topic_data.model_dump())
                loader.bulk_load(topic_data.model_dump(), table_name='default.ugc')

            extractor.confirm()
            state_instance.set_state("event_etl_data", None)
            logger.info("Data loaded")
        else:
            logger.info("No data to process")

    def run(self):
        while not self.killer.kill_now:
            self.run_etl()
            sleep(1)

        self.message_broker.disconnect()
        self.db_client.disconnect()
        logger.info("ETL process finished")


if __name__ == "__main__":
    logging.basicConfig(level=logging.DEBUG)
    etl_process = ETLProcess()
    etl_process.run()
