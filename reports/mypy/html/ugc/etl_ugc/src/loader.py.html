import backoff
from client.base.base_column_db import IColumnDBService


class Loader:
    def __init__(self, db_client: IColumnDBService):
        self.db_client = db_client

    @backoff.on_exception(backoff.expo, Exception, max_tries=5)
    def bulk_load(self, data: list, table_name: str) -> None:
        self.db_client.insert(table_name, data)
