"""Общие настройки проекта ugc_api."""

import os

from pydantic import Field
from pydantic_settings import BaseSettings


class Settings(BaseSettings):
    project_name: str = Field("event_etl", env="PROJECT_NAME")

    base_dir: str = Field(
        os.path.dirname(os.path.dirname(os.path.abspath(__file__))),
        env="BASE_DIR",
    )
    console_log_level: str = Field("DEBUG", env="CONSOLE_LOG_LEVEL")
    log_level: str = Field("DEBUG", env="LOG_LEVEL")


class ETLSettings(BaseSettings):
    batch_size: int = 10000
    broker_identifier: str = "event_etl"
    storage_key: str = "event_etl_data"
    lock_key: str = "event_etl_lock"

    class Config:
        env_prefix = "EVENT_ETL_"


class ClickHouseSettings(BaseSettings):
    host: str = "clickhouse-node1"
    port: int = 9000
    user: str = "default"
    password: str = ""
    database: str = "default"

    class Config:
        env_prefix = "CLICKHOUSE_"


class KafkaSettings(BaseSettings):
    bootstrap_servers: list[str] = ["kafka:9092"]
    topics: list[str] = ["events"]

    class Config:
        env_prefix = "KAFKA_"


class RedisSettings(BaseSettings):
    host: str = 'redis'
    port: int = 6379
    storage_name: str = "event_etl"

    class Config:
        env_prefix = "REDIS_"


kafka_settings = KafkaSettings()
clickhouse_settings = ClickHouseSettings()
etl_settings = ETLSettings()
redis_settings = RedisSettings()
settings = Settings()
