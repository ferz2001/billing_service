import logging
from collections import defaultdict

import backoff
from kafka import KafkaConsumer, errors
from client.base.base_msg_broker import IMessageBrokerConsumer

logger = logging.getLogger(__name__)


class KafkaMessageBrokerConsumer(IMessageBrokerConsumer):
    def __init__(self):
        self.consumer: KafkaConsumer | None = None
        self.batch_size: int | None = None
        self.max_offsets: dict[str | tuple, int] = {}

    def connect(
        self,
        host: str,
        port: int,
        identifier: str,
        batch_size: int = 10000,
        **kwargs,
    ):
        if self.consumer and self.consumer.bootstrap_connected():
            logger.warning("Consumer already connected")
            return

        self.batch_size = batch_size
        auto_offset_reset = kwargs.get("auto_offset_reset", "earliest")

        self.consumer = KafkaConsumer(
            bootstrap_servers=f"{host}:{port}",
            group_id=identifier,
            auto_offset_reset=auto_offset_reset,
            enable_auto_commit=False,
            max_poll_records=self.batch_size,
            **kwargs,
        )

    def disconnect(self):
        if self.consumer:
            self.consumer.close()
            logger.debug("Consumer disconnected")
        else:
            logger.debug("Can't disconnect, consumer wasn't connected.")

    def subscribe(self, channels: list[str], **kwargs):
        if self.consumer:
            self.consumer.subscribe(channels)
            logger.debug(f"Subscribed to topics: {channels}")

    @backoff.on_exception(
        backoff.expo,
        (errors.KafkaTimeoutError, errors.KafkaConnectionError),
        max_tries=5,
    )
    def poll(self) -> dict | None:
        if self.consumer:
            partitions = self.consumer.poll()
            event_data = defaultdict(list)
            for part in partitions:
                for record in partitions[part]:
                    event_data[part.topic].append(record.value)
                    self.max_offsets[(part.topic, part.partition)] = max(
                        self.max_offsets.get((part.topic, part.partition), -1),
                        record.offset,
                    )
            return event_data
        return None

    @backoff.on_exception(
        backoff.expo,
        (errors.KafkaTimeoutError, errors.KafkaConnectionError),
        max_tries=5,
    )
    def commit(self):
        if self.consumer:
            self.consumer.commit()
            logger.debug("Confirmed acquiring messages to the broker")
