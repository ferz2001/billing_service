from typing import Any

from logging import config as logging_config
import logging
from core.logger import LOGGING
import backoff
from clickhouse_driver import Client, errors
from client.base.base_column_db import IColumnDBService

logging_config.dictConfig(LOGGING)

logger = logging.getLogger(__name__)


class ClickHouseColumnDBClient(IColumnDBService):
    def __init__(self, host: str, port: int, user: str, password: str, database: str):
        self._db_url = f"clickhouse://{user}:{password}@{host}:{port}/{database}"
        self._client: Client | None = None

    def connect(self) -> bool:
        """Establish a connection with the database."""
        try:
            self._client = Client.from_url(self._db_url)
            return True
        except Exception as e:
            logger.info(f"Failed to connect to the database: {e}")
            return False

    def disconnect(self) -> bool:
        """Disconnect from the database."""
        try:
            if self._client:
                self._client.disconnect()
                self._client = None
            return True
        except Exception as e:
            logger.error(f"Failed to disconnect from the database: {e}")
            return False

    @backoff.on_exception(
        backoff.expo,
        errors.NetworkError,
        max_tries=5,
    )
    def insert(self, table: str, data: list[dict[str, Any]], columns: list[str] | None = None) -> bool:
        """Insert a bulk of data into a table."""
        try:
            if not columns:
                columns = list(data[0].keys())
            column_names = ", ".join(columns)
            logger.info(column_names)
            query = f"INSERT INTO {table} ({column_names}) VALUES"
            logger.info(query)
            if self._client:
                logger.info(f'Inserting data... with data {data}')
                res = self._client.execute(query, data)
                logger.info(res)
            return True
        except Exception as e:
            logger.error(f"Failed to insert data into the table: {e}")
            return False
