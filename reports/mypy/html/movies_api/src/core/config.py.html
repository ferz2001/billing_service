import os

from pydantic import Field
from pydantic_settings import BaseSettings


class Settings(BaseSettings):
    project_name: str = "Some project name"
    redis_host: str = Field('127.0.0.1', alias="REDIS_HOST")
    auth_url: str = Field('', alias="AUTH_SERVICE_URL")
    redis_port: str | int = Field(6379, alias="REDIS_PORT")
    elastic_host: str = Field('elastic', alias="ELASTIC_HOST")
    elastic_port: str | int = Field(9200, alias="ELASTIC_PORT")
    base_dir: str = os.path.dirname(os.path.dirname(os.path.abspath(__file__)))


cfg = Settings()
