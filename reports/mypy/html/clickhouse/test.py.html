import random
from datetime import datetime

from clickhouse_driver import Client
from pydantic import BaseModel


class UserEvent(BaseModel):
    user_id: str
    movie_id: str
    movie_timestamp: datetime
    event_time: datetime
    value: str


class ClickHouseUgc:
    def __init__(self, host: str = 'localhost'):
        self.client = Client(host=host)

    def bulk_insert(self, dataset: list[UserEvent]):
        values = ', '.join([
            f"('{event.user_id}', "
            f"'{event.movie_id}', "
            f"toDateTime('{event.movie_timestamp}'), "
            f"toDateTime('{event.event_time}'), "
            f"'{event.value}')"
            for event in dataset
        ])
        query = f"""
            INSERT INTO default.ugc (
            user_id, 
            movie_id, 
            movie_timestamp, 
            event_time, 
            value
            )
            VALUES 
            {values}
        """
        self.client.execute(query)

    def count_all_ugc(self):
        return self.client.execute('SELECT COUNT(*) FROM default.ugc')


def test():
    ch = ClickHouseUgc()

    dataset = []
    for i in range(0, 1000):
        dataset.append(
            UserEvent(
                user_id=str(random.randint(1, 1000000)),
                movie_id="111",
                movie_timestamp=datetime.now(),
                event_time=datetime.now(),
                value="delete"
            )
        )

    ch.bulk_insert(dataset)
    print(ch.count_all_ugc())


if __name__ == '__main__':
    test()
